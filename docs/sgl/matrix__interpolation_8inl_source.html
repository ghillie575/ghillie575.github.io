133 GLM_FUNC_QUALIFIER mat<4, 4, T, Q> interpolate(mat<4, 4, T, Q> const& m1, mat<4, 4, T, Q> const& m2, T const delta)
134 {
135 mat<4, 4, T, Q> m1rot = extractMatrixRotation(m1);
136 mat<4, 4, T, Q> dltRotation = m2 * transpose(m1rot);
137 vec<3, T, Q> dltAxis;
138 T dltAngle;
139 axisAngle(dltRotation, dltAxis, dltAngle);
140 mat<4, 4, T, Q> out = axisAngleMatrix(dltAxis, dltAngle * delta) * m1rot;
141 out[3][0] = m1[3][0] + delta * (m2[3][0] - m1[3][0]);
142 out[3][1] = m1[3][1] + delta * (m2[3][1] - m1[3][1]);
143 out[3][2] = m1[3][2] + delta * (m2[3][2] - m1[3][2]);
144 return out;
145 }
