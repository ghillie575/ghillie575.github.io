21 GLM_FUNC_QUALIFIER std::string format(const char* message, ...) {
22 std::size_t const STRING_BUFFER(4096);
23
24 assert(message != NULL);
25 assert(strlen(message) < STRING_BUFFER);
26
27 char buffer[STRING_BUFFER];
28 va_list list;
29
30#if GLM_COMPILER & GLM_COMPILER_CLANG
31# pragma clang diagnostic push
32# pragma clang diagnostic ignored "-Wformat-nonliteral"
33#endif
34
35 va_start(list, message);
36 vsnprintf(buffer, STRING_BUFFER, message, list);
37 va_end(list);
38
39#if GLM_COMPILER & GLM_COMPILER_CLANG
40# pragma clang diagnostic pop
41#endif
42
43 return buffer;
44 }
